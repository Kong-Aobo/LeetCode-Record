/* O(n^2) 但实际上比下面那种解法快很多
 * 遍历每个元素
 * 从该元素开始每次向两边扩展1，判断两侧元素是否相等
 * 直到不相等为止 快就快在少了很多非回文区间的验证 */
class Solution {
public:
    string longestPalindrome(string s) {
        int idx = 0, start = 0, len = 1;
        while(idx < s.size()){
            int left = idx - 1;
            while(idx < s.size() && s[idx] == s[idx+1])
                idx++;  //该循环非常巧妙 不仅解决了长度为偶数的回文字符串 还可优化整个问题
            int right = idx + 1;
            while(right < s.size() && left >= 0 && s[left] == s[right]){
                left--；
                right++;
            }
            if(right - left - 1 > len){
                start = left + 1;
                len = right - left - 1;
            }
            idx++;
        }
        return s.substr(start,len);
    }
};

/* O(n^2)
 * 区间DP: dp[i][j] = (s[i] == s[j]) && dp[i+1][j-1] ? true : false */
class Solution {
public:
    string longestPalindrome(string s) {
        int l = s.size();
        bool dp[l][l];
        memset(dp, 0, sizeof(dp));
        int start = 1, end = 0;
        for(int i = 0; i < l; i++)  //如果i=n-1向前遍历 dp[j][i]就变成dp[i][j]了
        {                           //传统区间DP枚举区间长度也可以
            for(int j = i; j >= 0; j--)
            {
                if(s[i] == s[j] && (i - j < 2 || dp[j + 1][i - 1]))
                    dp[j][i] = true;
                if(dp[j][i] && i - j + 1 > end - start + 1)
                {
                    start = j;
                    end = i;
                }
            }
        }
        return s.substr(start, end - start + 1);
    }
};
